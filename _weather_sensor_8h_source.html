66 // Sensor Types
67 // 0 - Weather Station 5-in-1; PN 7002510..12/7902510..12
68 // 1 - Weather Station 6-in-1; PN 7002585
69 // 2 - Thermo-/Hygro-Sensor 6-in-1; PN 7009999
70 // 4 - Soil Moisture Sensor 6-in-1; PN 7009972
71 // 9 - Professional Rain Gauge (5-in-1 decoder)
72 // ? - Air Quality Sensor
73 // ? - Water Leakage Sensor
74 // ? - Pool Thermometer
75 // ? - Lightning Sensor

77 #define SENSOR_TYPE_WEATHER1 1 // Weather Station
78 #define SENSOR_TYPE_THERMO_HYGRO 2 // Thermo-/Hygro-Sensor
79 #define SENSOR_TYPE_SOIL 4 // Soil Temperature and Moisture (from 6-in-1 decoder)
80 #define SENSOR_TYPE_RAIN 9 // Professional Rain Gauge (from 5-in-1 decoder)
